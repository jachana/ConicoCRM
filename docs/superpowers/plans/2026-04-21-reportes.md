def _validate_dates(date_from: date, date_to: date) -> None:

# ── Ventas helpers ────────────────────────────────────────────────────────────

def _get_ventas(date_from: date, date_to: date, db: Session, vendedor_id: int | None) -> dict:
    q = (
        db.query(Factura)
        .options(joinedload(Factura.cliente), joinedload(Factura.vendedor))
        .filter(
            Factura.fecha >= date_from,
            Factura.fecha <= date_to,
            Factura.estado != "anulada",
        )
    )
    if vendedor_id:
        q = q.filter(Factura.vendedor_id == vendedor_id)
    facturas = q.all()

    total_vendido = sum(float(f.total) for f in facturas)
    num_facturas = len(facturas)
    ticket_promedio = total_vendido / num_facturas if num_facturas else 0

    pagos_q = db.query(func.sum(Pago.monto)).filter(
        Pago.factura_id.in_([f.id for f in facturas])
    ).scalar() or 0
    total_por_cobrar = total_vendido - float(pagos_q)

    # Variación vs período anterior de igual duración
    duracion = (date_to - date_from).days + 1
    prev_to = date_from - timedelta(days=1)
    prev_from = prev_to - timedelta(days=duracion - 1)
    prev_q = db.query(func.sum(Factura.total)).filter(
        Factura.fecha >= prev_from,
        Factura.fecha <= prev_to,
        Factura.estado != "anulada",
    )
    if vendedor_id:
        prev_q = prev_q.filter(Factura.vendedor_id == vendedor_id)
    prev_total = float(prev_q.scalar() or 0)
    variacion = round((total_vendido - prev_total) / prev_total * 100, 1) if prev_total else 0.0

    # Ventas diarias
    daily: dict[str, float] = {}
    for f in facturas:
        k = f.fecha.isoformat()
        daily[k] = daily.get(k, 0) + float(f.total)
    ventas_diarias = [{"fecha": k, "monto": v} for k, v in sorted(daily.items())]

    # Top clientes
    cliente_map: dict[int, dict] = {}
    for f in facturas:
        if not f.cliente_id:
            continue
        if f.cliente_id not in cliente_map:
            cliente_map[f.cliente_id] = {
                "cliente_id": f.cliente_id,
                "nombre": f.cliente.nombre if f.cliente else "",
                "total": 0.0,
                "num_facturas": 0,
            }
        cliente_map[f.cliente_id]["total"] += float(f.total)
        cliente_map[f.cliente_id]["num_facturas"] += 1
    top_clientes = sorted(cliente_map.values(), key=lambda x: x["total"], reverse=True)[:10]

    # Por vendedor
    vendedor_map: dict[int, dict] = {}
    for f in facturas:
        if not f.vendedor_id:
            continue
        if f.vendedor_id not in vendedor_map:
            vendedor_map[f.vendedor_id] = {
                "vendedor_id": f.vendedor_id,
                "nombre": f.vendedor.name if f.vendedor else "",
                "total": 0.0,
                "num_facturas": 0,
            }
        vendedor_map[f.vendedor_id]["total"] += float(f.total)
        vendedor_map[f.vendedor_id]["num_facturas"] += 1
    por_vendedor = sorted(vendedor_map.values(), key=lambda x: x["total"], reverse=True)

    return {
        "kpis": {
            "total_vendido": round(total_vendido, 2),
            "num_facturas": num_facturas,
            "ticket_promedio": round(ticket_promedio, 2),
            "total_por_cobrar": round(total_por_cobrar, 2),
            "variacion_vs_periodo_anterior": variacion,
        },
        "ventas_diarias": ventas_diarias,
        "top_clientes": top_clientes,
        "por_vendedor": por_vendedor,
    }


@router.get("/ventas")
def get_ventas(
    date_from: Annotated[date, Query()],
    date_to: Annotated[date, Query()],

):

    vendedor_id = user.id if user.role == "vendedor" else None
    return _get_ventas(date_from, date_to, db, vendedor_id)


# ── Cobranza helpers ──────────────────────────────────────────────────────────

def _get_cobranza(date_from: date, date_to: date, db: Session) -> dict:
    facturas = (
        db.query(Factura)
        .options(joinedload(Factura.empresa))
        .filter(
            Factura.fecha >= date_from,
            Factura.fecha <= date_to,
            Factura.estado != "anulada",
        )
        .all()
    )

    today = date.today()
    total_por_cobrar = 0.0
    total_vencido = 0.0
    proximas_7d = 0.0
    buckets = {
        "d_0_30": {"count": 0, "monto": 0.0},
        "d_31_60": {"count": 0, "monto": 0.0},
        "d_61_90": {"count": 0, "monto": 0.0},
        "d_90_plus": {"count": 0, "monto": 0.0},
    }
    empresa_map: dict[int, dict] = {}

    for f in facturas:
        pagado = float(f.monto_pagado or 0)
        saldo = float(f.total) - pagado
        if saldo <= 0:
            continue
        total_por_cobrar += saldo

        if not f.fecha_vencimiento:
            continue

        dias_hasta = (f.fecha_vencimiento - today).days
        if 0 <= dias_hasta <= 7:
            proximas_7d += saldo

        if f.fecha_vencimiento < today:
            dias_vencida = (today - f.fecha_vencimiento).days
            total_vencido += saldo
            if dias_vencida <= 30:
                bk = "d_0_30"
            elif dias_vencida <= 60:
                bk = "d_31_60"
            elif dias_vencida <= 90:
                bk = "d_61_90"
            else:
                bk = "d_90_plus"
            buckets[bk]["count"] += 1
            buckets[bk]["monto"] += saldo

            if f.empresa_id:
                if f.empresa_id not in empresa_map:
                    empresa_map[f.empresa_id] = {
                        "empresa_id": f.empresa_id,
                        "nombre": f.empresa.nombre if f.empresa else "",
                        "saldo": 0.0,
                        "dias_vencida": dias_vencida,
                    }
                empresa_map[f.empresa_id]["saldo"] += saldo
                empresa_map[f.empresa_id]["dias_vencida"] = max(
                    empresa_map[f.empresa_id]["dias_vencida"], dias_vencida
                )

    por_empresa = sorted(empresa_map.values(), key=lambda x: x["saldo"], reverse=True)

    return {
        "kpis": {
            "total_por_cobrar": round(total_por_cobrar, 2),
            "total_vencido": round(total_vencido, 2),
            "proximas_a_vencer_7d": round(proximas_7d, 2),
        },
        "aging": buckets,
        "por_empresa": por_empresa,
    }


@router.get("/cobranza")
def get_cobranza(
    date_from: Annotated[date, Query()],
    date_to: Annotated[date, Query()],
    perms: tuple[User, Session] = Depends(require_permission("facturas", "view")),
):
    _, db = perms
    _validate_dates(date_from, date_to)
    return _get_cobranza(date_from, date_to, db)


# ── Inventario helpers ────────────────────────────────────────────────────────

def _get_inventario(date_from: date, date_to: date, db: Session) -> dict:
    productos = db.query(Producto).all()

    valor_total = sum(
        float(p.precio_costo or 0) * float(p.stock_actual or 0) for p in productos
    )
    bajo_minimo = [
        {
            "producto_id": p.id,
            "nombre": p.nombre,
            "sku": p.sku or "",
            "stock_actual": float(p.stock_actual or 0),
            "stock_minimo": float(p.stock_minimo or 0),
        }
        for p in productos
        if p.stock_minimo and (p.stock_actual or 0) < p.stock_minimo
    ]
    sin_stock = [p for p in productos if (p.stock_actual or 0) <= 0]

    # Top vendidos from movimientos_inventario in period
    movs = (
        db.query(MovimientoInventario)
        .options(joinedload(MovimientoInventario.producto))
        .filter(
            MovimientoInventario.tipo == "salida",
            MovimientoInventario.created_at >= date_from,
            MovimientoInventario.created_at < date_to + timedelta(days=1),
        )
        .all()
    )
    top_map: dict[int, dict] = {}
    for m in movs:
        if not m.producto_id:
            continue
        if m.producto_id not in top_map:
            top_map[m.producto_id] = {
                "producto_id": m.producto_id,
                "nombre": m.producto.nombre if m.producto else "",
                "cantidad_vendida": 0.0,
                "monto_total": 0.0,
            }
        top_map[m.producto_id]["cantidad_vendida"] += float(m.cantidad or 0)
        top_map[m.producto_id]["monto_total"] += float(m.cantidad or 0) * float(
            m.producto.precio_venta or 0 if m.producto else 0
        )
    top_vendidos = sorted(top_map.values(), key=lambda x: x["cantidad_vendida"], reverse=True)[:20]

    return {
        "kpis": {
            "valor_total_stock": round(valor_total, 2),
            "num_bajo_minimo": len(bajo_minimo),
            "num_sin_stock": len(sin_stock),
        },
        "bajo_minimo": bajo_minimo,
        "top_vendidos": top_vendidos,
    }

def _get_margenes(date_from: date, date_to: date, db: Session, vendedor_id: int | None) -> dict:
    fq = (
        db.query(Factura)
        .options(joinedload(Factura.lineas))
        .filter(
            Factura.fecha >= date_from,
            Factura.fecha <= date_to,
            Factura.estado != "anulada",
        )
    )
    if vendedor_id:
        fq = fq.filter(Factura.vendedor_id == vendedor_id)
    facturas = fq.all()

    producto_map: dict[str, dict] = {}
    por_factura = []
    all_margenes_pct = []

    for f in facturas:
        lineas = f.lineas or []
        f_neto = float(f.total_neto or f.total or 0)
        f_margen = 0.0
        for ln in lineas:
            if ln.margen is None:
                continue
            margen_val = float(ln.margen)
            desc = ln.descripcion or ""
            if desc not in producto_map:
                producto_map[desc] = {
                    "producto_id": ln.producto_id,
                    "nombre": desc,
                    "cantidad_vendida": 0.0,
                    "precio_costo_promedio": 0.0,
                    "precio_venta_promedio": 0.0,
                    "margen_pct": 0.0,
                    "_margen_sum": 0.0,
                    "_count": 0,
                }
            producto_map[desc]["cantidad_vendida"] += float(ln.cantidad or 0)
            producto_map[desc]["_margen_sum"] += margen_val
            producto_map[desc]["_count"] += 1
            if ln.valor_neto and ln.cantidad:
                pv = float(ln.valor_neto) / float(ln.cantidad)
                producto_map[desc]["precio_venta_promedio"] = pv
            f_margen += margen_val * float(ln.cantidad or 1)

        if f_neto > 0:
            f_margen_pct = round(f_margen / f_neto * 100, 1)
        else:
            f_margen_pct = 0.0
        all_margenes_pct.append(f_margen_pct)
        por_factura.append({
            "factura_id": f.id,
            "numero": f.numero,
            "total": float(f.total),
            "margen_total": round(f_margen, 2),
            "margen_pct": f_margen_pct,
        })

    # Compute per-product margen_pct
    productos_list = []
    for p in producto_map.values():
        if p["_count"] > 0:
            p["margen_pct"] = round(p["_margen_sum"] / p["_count"] * 100, 1)
        del p["_margen_sum"]
        del p["_count"]
        productos_list.append(p)

    productos_list.sort(key=lambda x: x["margen_pct"], reverse=True)

    margen_promedio = round(sum(all_margenes_pct) / len(all_margenes_pct), 1) if all_margenes_pct else 0.0
    mejor = productos_list[0] if productos_list else None
    peor = productos_list[-1] if productos_list else None

    return {
        "kpis": {
            "margen_promedio_pct": margen_promedio,
            "mejor_producto": {"nombre": mejor["nombre"], "margen_pct": mejor["margen_pct"]} if mejor else None,
            "peor_producto": {"nombre": peor["nombre"], "margen_pct": peor["margen_pct"]} if peor else None,
        },
        "por_producto": productos_list[:20],
        "por_factura": por_factura,
    }

from app.models.dte import DteEmision


def _get_dte(date_from: date, date_to: date, db: Session) -> dict:
    emisiones = (
        db.query(DteEmision)
        .filter(
            DteEmision.created_at >= date_from,
            DteEmision.created_at < date_to + timedelta(days=1),
        )
        .all()
    )

    total = len(emisiones)
    aceptadas = sum(1 for e in emisiones if e.estado == "aceptada")
    rechazadas = sum(1 for e in emisiones if e.estado == "rechazada")
    pendientes = sum(1 for e in emisiones if e.estado in ("pendiente", "procesando"))

    tipo_map: dict[str, dict] = {}
    for e in emisiones:
        t = e.tipo
        if t not in tipo_map:
            labels = {"033": "Factura", "061": "Nota Crédito", "056": "Nota Débito"}
            tipo_map[t] = {"tipo": t, "label": labels.get(t, t), "count": 0, "aceptadas": 0}
        tipo_map[t]["count"] += 1
        if e.estado == "aceptada":
            tipo_map[t]["aceptadas"] += 1

    emisiones_list = [
        {
            "id": e.id,
            "tipo": e.tipo,
            "folio": e.folio,
            "estado": e.estado,
            "monto_total": e.monto_total,
            "created_at": e.created_at.date().isoformat() if e.created_at else None,
            "detalle_rechazo": e.respuesta_sii.get("detalle") if e.respuesta_sii and e.estado == "rechazada" else None,
        }
        for e in emisiones
    ]

    return {
        "kpis": {
            "total_emitidos": total,
            "aceptadas": aceptadas,
            "rechazadas": rechazadas,
            "pendientes": pendientes,
        },
        "por_tipo": list(tipo_map.values()),
        "emisiones": emisiones_list,
    }

function startOfMonth(d: Date) {

function formatISO(d: Date) {

function fmt(n: number) {

function KpiCard({ label, value, sub, subColor }: { label: string; value: string; sub?: string; subColor?: string }) {
  return (

async function exportFile(tab: Tab, format: 'excel' | 'pdf', dateFrom: string, dateTo: string) {
  const response = await api.get(
    `/api/reportes/${tab}/export/${format}?date_from=${dateFrom}&date_to=${dateTo}`,
    { responseType: 'blob' }
  )
  const ext = format === 'excel' ? 'xlsx' : 'pdf'
  const url = URL.createObjectURL(new Blob([response.data]))
  const a = document.createElement('a')
  a.href = url
  a.download = `${tab}-${dateFrom}-${dateTo}.${ext}`
  a.click()
  URL.revokeObjectURL(url)
}

const TABS: { key: Tab; label: string }[] = [
  { key: 'ventas', label: 'Ventas' },
  { key: 'cobranza', label: 'Cobranza' },
  { key: 'inventario', label: 'Inventario' },
  { key: 'compras', label: 'Compras' },
  { key: 'margenes', label: 'Márgenes' },
  { key: 'dte', label: 'DTE' },
]

function getPresetDates(preset: Preset): { from: string; to: string } {
  const today = new Date()
  const to = formatISO(today)
  if (preset === 'este_mes') {
    return { from: formatISO(startOfMonth(today)), to }
  }
  if (preset === 'mes_anterior') {
    const lastMonth = new Date(today.getFullYear(), today.getMonth() - 1, 1)
    const lastMonthEnd = new Date(today.getFullYear(), today.getMonth(), 0)
    return { from: formatISO(lastMonth), to: formatISO(lastMonthEnd) }
  }
  if (preset === 'este_anio') {
    return { from: `${today.getFullYear()}-01-01`, to }
  }
  if (preset === 'ultimos_3_meses') {
    const from = new Date(today)
    from.setMonth(from.getMonth() - 3)
    return { from: formatISO(from), to }
  }
  return { from: formatISO(startOfMonth(today)), to }
}

export default function Reportes() {
  const [activeTab, setActiveTab] = useState<Tab>('ventas')
  const [preset, setPreset] = useState<Preset>('este_mes')
  const initial = getPresetDates('este_mes')
  const [dateFrom, setDateFrom] = useState(initial.from)
  const [dateTo, setDateTo] = useState(initial.to)

  const handlePreset = useCallback((p: Preset) => {
    setPreset(p)
    if (p !== 'personalizado') {
      const { from, to } = getPresetDates(p)
      setDateFrom(from)
      setDateTo(to)
    }
  }, [])

  return (
    <div className="min-h-screen bg-[#0b1120] text-gray-100">
      {/* Header */}
      <div className="px-6 pt-5 pb-0 flex items-center justify-between border-b border-white/[0.08]">
        <h1 className="text-lg font-semibold text-white">Reportes</h1>
        <div className="flex gap-2 items-center pb-4">
          <select
            value={preset}
            onChange={e => handlePreset(e.target.value as Preset)}
            className="bg-gray-900 border border-white/[0.1] text-gray-300 px-3 py-1.5 rounded-lg text-xs"
          >
            <option value="este_mes">Este mes</option>
            <option value="mes_anterior">Mes anterior</option>
            <option value="este_anio">Este año</option>
            <option value="ultimos_3_meses">Últimos 3 meses</option>
            <option value="personalizado">Rango personalizado</option>
          </select>
          {preset === 'personalizado' && (
            <>
              <input
                type="date"
                value={dateFrom}
                onChange={e => setDateFrom(e.target.value)}
                className="bg-gray-900 border border-white/[0.1] text-gray-300 px-3 py-1.5 rounded-lg text-xs"
              />
              <span className="text-gray-600">→</span>
              <input
                type="date"
                value={dateTo}
                onChange={e => setDateTo(e.target.value)}
                className="bg-gray-900 border border-white/[0.1] text-gray-300 px-3 py-1.5 rounded-lg text-xs"
              />
            </>
          )}
        </div>
      </div>

      {/* Tabs */}
      <div className="flex border-b border-white/[0.08] px-6">
        {TABS.map(t => (
          <button
            key={t.key}
            onClick={() => setActiveTab(t.key)}
            className={`px-4 py-2.5 text-xs font-medium border-b-2 transition-colors ${
              activeTab === t.key
                ? 'border-amber-400 text-amber-400'
                : 'border-transparent text-gray-500 hover:text-gray-300'
            }`}
          >
            {t.label}
          </button>
        ))}
      </div>

      {/* Tab content */}
      <div className="p-6">
        {activeTab === 'ventas'     && <TabVentas     dateFrom={dateFrom} dateTo={dateTo} />}
        {activeTab === 'cobranza'   && <TabCobranza   dateFrom={dateFrom} dateTo={dateTo} />}
        {activeTab === 'inventario' && <TabInventario dateFrom={dateFrom} dateTo={dateTo} />}
        {activeTab === 'compras'    && <TabCompras    dateFrom={dateFrom} dateTo={dateTo} />}
        {activeTab === 'margenes'   && <TabMargenes   dateFrom={dateFrom} dateTo={dateTo} />}
        {activeTab === 'dte'        && <TabDte        dateFrom={dateFrom} dateTo={dateTo} />}
      </div>
    </div>
  )
}
